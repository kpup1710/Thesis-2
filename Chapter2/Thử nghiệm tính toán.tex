\begin{equation}
    \label{vdchap2}
    \begin{aligned}
        \min\ & f(x) = \dfrac{x_1^2 + x_2^2 +3}{1 + 2x_1 + 8x_2}\\
         s.t\ & g(x) = -x_1^2 - 2x_1x_2 \leq -4,\\
              & x_1, x_2 \geq 0.
    \end{aligned}
\end{equation}
